#include <iostream>
using namespace std;

int arr1[4] = { 1,5,4,9 };
int arr2[13] = { 1,1,1,5,1,5,4,9,1,5,4,9 };

int isSame(int idx) {
	for (int i = 0; i < 4; i++) {
		if (arr1[i] != arr2[idx + i]) return 0;
	}
	return 1;
}

int main() {
	int cnt = 0;
	for (int i = 0; i < 10; i++) {
		if (isSame(i)) cnt++;
	}

	cout << cnt << "°³";
	
	return 0;
}
